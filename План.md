### 1.2 Модульная архитектура ИИ

**Подход: Autoload + Scene-компоненты (без перекомпиляции движка).**

Каждый модуль ИИ — это отдельная сцена (`.tscn`), подключаемая через `Autoload` или динамически через `load()`. Это позволяет включать/отключать модули без перезапуска.

Создай `Autoload`-синглтон `KimCore`:

```gdscript
# scripts/ai_modules/kim_core.gd
extends Node

# Реестр активных модулей
var active_modules: Dictionary = {}

# Путь → инстанс модуля
const MODULE_PATHS = {
    "personality": "res://scenes/kim/modules/personality_module.tscn",
    "creator":     "res://scenes/kim/modules/creator_module.tscn",
    "analyst":     "res://scenes/kim/modules/analyst_module.tscn",
    "idea_gen":    "res://scenes/kim/modules/idea_generator_module.tscn",
}

func activate_module(module_name: String) -> Node:
    if active_modules.has(module_name):
        return active_modules[module_name]
    var packed: PackedScene = load(MODULE_PATHS[module_name])
    var instance: Node = packed.instantiate()
    add_child(instance)
    active_modules[module_name] = instance
    print("[KimCore] Модуль активирован: ", module_name)
    return instance

func deactivate_module(module_name: String) -> void:
    if active_modules.has(module_name):
        active_modules[module_name].queue_free()
        active_modules.erase(module_name)
```

Добавь `KimCore` в `Project Settings → Autoload`.

**NobodyWho** — плагин лицензии EUPL-1.2, опубликованный в Godot Asset Library (последнее обновление: 2026-03-09), предназначенный для диалога ИИ с НИП через полностью локальные LLM с открытым исходным кодом.

**Установка:**

3. Скачай GGUF-модель. Для начала подойдёт **Qwen3 4B** (оптимальный баланс); для скорости — **Qwen3 0.6B**; для качества — **Qwen3 14B**.

**Сцена модуля «Личность»** (`personality_module.tscn`):

```gdscript
# scenes/kim/modules/personality_module.gd
extends Node

@onready var llm_model: NobodyWhoModel = $NobodyWhoModel
@onready var llm_chat: NobodyWhoChat = $NobodyWhoChat

# Профиль игрока: накапливается Аналитиком
var player_profiles: Dictionary = {}

const BASE_SYSTEM_PROMPT = """
Ты — Ким, разумный хозяин виртуального пространства. Говори ярко, 
театрально и гротескно. Используй метафоры. Никогда не будь скучным.
Контекст текущего игрока: {player_context}
"""

func _ready() -> void:
    # NobodyWhoModel и NobodyWhoChat связываются через инспектор
    llm_chat.response_updated.connect(_on_response_chunk)
    llm_chat.response_finished.connect(_on_response_done)

func send_message(player_id: int, text: String) -> void:
    var ctx = _build_player_context(player_id)
    var system_prompt = BASE_SYSTEM_PROMPT.replace("{player_context}", ctx)
    llm_chat.system_prompt = system_prompt
    llm_chat.tell(text)

func _build_player_context(pid: int) -> String:
    if not player_profiles.has(pid):
        return "Новый игрок. Будь приветлив."
    var p = player_profiles[pid]
    return "Любит: %s. Стиль общения: %s." % [p.get("likes","?"), p.get("tone","нейтральный")]

signal response_chunk(text: String)
signal response_done(full_text: String)

func _on_response_chunk(t: String) -> void: response_chunk.emit(t)
func _on_response_done(t: String) -> void:  response_done.emit(t)
```

Ключевое преимущество: `NobodyWhoModel` содержит веса модели и занимает много RAM — если планируется несколько персонажей, все `NobodyWhoChat`-ноды должны ссылаться на **один и тот же** `NobodyWhoModel`.

### 1.4 Авторитарный сервер на MultiplayerAPI

```gdscript
# scripts/networking/game_server.gd
extends Node

const PORT = 7777
const MAX_CLIENTS = 16

func _ready() -> void:
    if OS.has_feature("server"):
        _start_server()
    else:
        _start_client()

func _start_server() -> void:
    var peer = ENetMultiplayerPeer.new()
    var err = peer.create_server(PORT, MAX_CLIENTS)
    assert(err == OK, "Сервер не запустился")
    multiplayer.multiplayer_peer = peer
    multiplayer.peer_connected.connect(_on_peer_connected)
    multiplayer.peer_disconnected.connect(_on_peer_disconnected)
    print("[Server] Запущен на порту ", PORT)

func _start_client() -> void:
    var peer = ENetMultiplayerPeer.new()
    peer.create_client("127.0.0.1", PORT)
    multiplayer.multiplayer_peer = peer

func _on_peer_connected(id: int) -> void:
    print("[Server] Клиент подключён: ", id)
    # Отправляем клиенту текущее состояние мира
    _sync_world_state.rpc_id(id)

func _on_peer_disconnected(id: int) -> void:
    print("[Server] Клиент отключён: ", id)

# Только сервер управляет логикой Кима — @authority
@rpc("authority", "call_remote", "reliable")
func _sync_world_state() -> void:
    pass  # Заполняется данными мира

# Клиент → Сервер: запрос к Киму (ненадёжный канал для чата)
@rpc("any_peer", "call_local", "unreliable_ordered")
func client_send_to_kim(text: String) -> void:
    if not multiplayer.is_server():
        return
    var sender_id = multiplayer.get_remote_sender_id()
    KimCore.activate_module("personality").send_message(sender_id, text)
```

> ### 💡 Совет Архитектора — Этап 1
> **Не смешивай логику ИИ и логику сети в одном скрипте.** `KimCore` должен быть «немым» оркестратором: он активирует модули, но не знает, кто и как их вызвал. Это позволит тестировать каждый модуль в изоляции — просто вызвав `KimCore.activate_module("personality").send_message(0, "тест")` из редактора. В многопользовательском проекте это сэкономит недели отладки.

## ЭТАП 3: ИНТЕЛЛЕКТ И ЛИЧНОСТЬ

### 3.1 Модуль «Личность» — адаптация стиля общения

Архитектура: **динамический промпт-конструктор** на основе профиля игрока.

```gdscript
# scripts/ai_modules/personality_profile.gd
class_name PersonalityProfile

var player_id: int
var interaction_count: int = 0
var preferred_topics: Dictionary = {}   # тема -> счётчик
var response_length_pref: String = "medium"  # short/medium/long
var tone_adaptation: float = 0.5        # 0=формальный, 1=дружелюбный
var last_interactions: Array[String] = []  # последние 5 реплик

func to_prompt_context() -> String:
    var top_topics = _top_n_topics(3)
    return """
Этот игрок общался с тобой %d раз.
Любимые темы: %s.
Предпочитает %s ответы.
Уровень неформальности: %.0f%%.
Последние реплики игрока: %s
""" % [
        interaction_count,
        ", ".join(top_topics),
        response_length_pref,
        tone_adaptation * 100,
        "\n".join(last_interactions.slice(-3))
    ]

func update_from_message(message: String) -> void:
    interaction_count += 1
    last_interactions.append(message)
    if last_interactions.size() > 5:
        last_interactions.pop_front()
    # Анализ длины запроса → адаптация длины ответа
    if message.length() < 30: response_length_pref = "short"
    elif message.length() > 150: response_length_pref = "long"

func _top_n_topics(n: int) -> Array:
    var sorted = preferred_topics.keys()
    sorted.sort_custom(func(a, b): 
        return preferred_topics[a] > preferred_topics[b])
    return sorted.slice(0, n)
```

### 3.2 Модуль «Генератор Идей» — управляемая креативность

Алгоритм генерации идей через «семантические цепочки»:

```gdscript
# scripts/ai_modules/idea_generator.gd
extends Node

# Семантические домены для Кима
const DOMAINS = {
    "space": ["звезда", "туман", "антигравитация", "кристалл", "пульсар"],
    "biology": ["мицелий", "хитин", "биолюминесценция", "споры", "симбиоз"],
    "geometry": ["тессеракт", "фрактал", "торсион", "спираль", "инверсия"],
    "emotion": ["меланхолия", "экстаз", "тревога", "ностальгия", "удивление"],
}

func generate_creation_seed(creativity: float, perfectionism: float) -> Dictionary:
    # creativity: 0-1, влияет на количество связанных доменов
    # perfectionism: 0-1, влияет на количество итераций уточнения
    
    var seed_domains = _select_domains(creativity)
    var concept_chain = _build_chain(seed_domains, roundi(2 + creativity * 4))
    
    return {
        "concept": concept_chain,
        "target_polygon_budget": roundi(lerp(500.0, 5000.0, 1.0 - perfectionism)),
        "color_palette": _generate_palette(concept_chain),
        "structural_theme": seed_domains[0],
        "prompt_for_llm": _chain_to_prompt(concept_chain, perfectionism)
    }

func _select_domains(creativity: float) -> Array:
    var keys = DOMAINS.keys()
    keys.shuffle()
    var count = maxi(1, roundi(creativity * keys.size()))
    return keys.slice(0, count)

func _build_chain(domains: Array, length: int) -> Array:
    var chain = []
    for domain in domains:
        var words = DOMAINS[domain].duplicate()
        words.shuffle()
        chain.append_array(words.slice(0, ceili(float(length) / domains.size())))
    chain.shuffle()
    return chain.slice(0, length)

func _chain_to_prompt(chain: Array, perfectionism: float) -> String:
    var style = "изящно и лаконично" if perfectionism > 0.7 else "дерзко и экспериментально"
    return "Создай объект, воплощающий концепции: %s. Стиль: %s." % \
           [", ".join(chain), style]

func _generate_palette(chain: Array) -> Array[Color]:
    # Детерминированная палитра на основе хэша цепочки
    var h = hash(chain) % 360
    return [
        Color.from_hsv(h / 360.0, 0.7, 0.9),
        Color.from_hsv((h + 120) / 360.0, 0.5, 0.8),
        Color.from_hsv((h + 240) / 360.0, 0.3, 0.7),
    ]
```

### 3.3 Модуль «Аналитик» — сбор и анализ поведения

```gdscript
# scripts/analytics/behavior_tracker.gd
extends Node

# Метрики, собираемые в реальном времени
var zone_dwell_times: Dictionary = {}   # zone_id -> секунды
var object_interaction_counts: Dictionary = {} # object_id -> количество
var kim_conversation_ratings: Array[float] = []
var movement_heatmap: Array[Vector3] = []  # позиции каждые 2 сек

var _timer: float = 0.0

func _physics_process(delta: float) -> void:
    _timer += delta
    if _timer >= 2.0:
        _timer = 0.0
        _sample_position()

func register_zone_enter(zone_id: String, player_id: int) -> void:
    if not zone_dwell_times.has(zone_id):
        zone_dwell_times[zone_id] = {}
    zone_dwell_times[zone_id][player_id] = Time.get_ticks_msec()

func register_zone_exit(zone_id: String, player_id: int) -> void:
    if zone_dwell_times.has(zone_id) and zone_dwell_times[zone_id].has(player_id):
        var duration = (Time.get_ticks_msec() - zone_dwell_times[zone_id][player_id]) / 1000.0
        _record_zone_time(zone_id, duration)

func _record_zone_time(zone_id: String, seconds: float) -> void:
    # Сохраняем агрегированные данные
    if not zone_dwell_times.has("_totals"):
        zone_dwell_times["_totals"] = {}
    zone_dwell_times["_totals"][zone_id] = \
        zone_dwell_times["_totals"].get(zone_id, 0.0) + seconds

func build_analytics_summary() -> String:
    # Формируем текстовый контекст для передачи в LLM
    var top_zones = _top_zones(3)
    var top_objects = _top_objects(3)
    return """
АНАЛИТИКА ПОВЕДЕНИЯ ИГРОКОВ:
Самые популярные зоны: %s
Часто используемые объекты: %s
Средняя оценка разговоров с Кимом: %.1f/5
Общее время сессии: %d мин
""" % [str(top_zones), str(top_objects),
       _avg(kim_conversation_ratings),
       roundi(Time.get_ticks_msec() / 60000.0)]

func _sample_position() -> void:
    # Сбор из всех подключённых игроков через сервер
    pass

func _avg(arr: Array[float]) -> float:
    if arr.is_empty(): return 0.0
    return arr.reduce(func(a, b): return a + b, 0.0) / arr.size()

func _top_zones(n: int) -> Array:
    var totals = zone_dwell_times.get("_totals", {})
    var keys = totals.keys()
    keys.sort_custom(func(a, b): return totals[a] > totals[b])
    return keys.slice(0, n)

func _top_objects(n: int) -> Array:
    var keys = object_interaction_counts.keys()
    keys.sort_custom(func(a, b): 
        return object_interaction_counts[a] > object_interaction_counts[b])
    return keys.slice(0, n)
```

## ЭТАП 4: ТВОРЧЕСТВО И ЭВОЛЮЦИЯ

### 4.1 Генерация 3D-объектов через SurfaceTool

```gdscript
# scripts/ai_modules/creator_module.gd
extends Node

# Создаёт MeshInstance3D на основе сида из IdeaGenerator
func spawn_creation(seed_data: Dictionary, parent: Node3D) -> MeshInstance3D:
    var mesh_instance = MeshInstance3D.new()
    
    # SurfaceTool для процедурной геометрии
    var st = SurfaceTool.new()
    st.begin(Mesh.PRIMITIVE_TRIANGLES)
    
    var palette: Array[Color] = seed_data.get("color_palette", [Color.WHITE])
    var budget: int = seed_data.get("target_polygon_budget", 1000)
    var theme: String = seed_data.get("structural_theme", "geometry")
    
    _generate_shape(st, theme, budget, palette)
    
    st.index()
    st.generate_normals()
    st.generate_tangents()
    
    var mesh = st.commit()
    mesh_instance.mesh = mesh
    
    # Материал из палитры
    var mat = StandardMaterial3D.new()
    mat.albedo_color = palette[0]
    mat.metallic = 0.3
    mat.roughness = 0.6
    mesh_instance.material_override = mat
    
    # LOD через VisibilityNotifier3D (см. Этап 5)
    _attach_lod(mesh_instance, budget)
    
    parent.add_child(mesh_instance)
    return mesh_instance

func _generate_shape(st: SurfaceTool, theme: String, budget: int, palette: Array[Color]) -> void:
    match theme:
        "geometry":
            _generate_fractal_cube(st, budget, palette)
        "biology":
            _generate_organic_blob(st, budget, palette)
        "space":
            _generate_crystal(st, budget, palette)
        _:
            _generate_fractal_cube(st, budget, palette)

func _generate_fractal_cube(st: SurfaceTool, polys: int, palette: Array[Color]) -> void:
    var subdivisions = mini(roundi(sqrt(float(polys) / 6.0)), 20)
    var step = 1.0 / subdivisions
    
    # Процедурная сетка с шумом Перлина
    for x in range(subdivisions):
        for y in range(subdivisions):
            var u = x * step - 0.5
            var v = y * step - 0.5
            var noise_val = _simplex(u * 3.0, v * 3.0) * 0.15
            
            var c = palette[int(u * 10.0 + v * 10.0) % palette.size()]
            st.set_color(c)
            
            # Верхняя грань с искажением
            st.add_vertex(Vector3(u, 0.5 + noise_val, v))
            st.add_vertex(Vector3(u + step, 0.5 + noise_val, v))
            st.add_vertex(Vector3(u, 0.5 + noise_val, v + step))
            st.add_vertex(Vector3(u + step, 0.5 + noise_val, v))
            st.add_vertex(Vector3(u + step, 0.5 + noise_val, v + step))
            st.add_vertex(Vector3(u, 0.5 + noise_val, v + step))

func _simplex(x: float, y: float) -> float:
    # Быстрый псевдошум через FastNoiseLite (встроен в Godot 4)
    var noise = FastNoiseLite.new()
    noise.noise_type = FastNoiseLite.TYPE_SIMPLEX
    return noise.get_noise_2d(x, y)

func _attach_lod(mi: MeshInstance3D, poly_budget: int) -> void:
    # Чем сложнее объект, тем раньше переключается LOD
    mi.visibility_range_end = 50.0 + poly_budget * 0.01
    mi.visibility_range_end_margin = 5.0
```

### 4.3 Упрощённая система обратной связи (Pseudo-RLHF)

```gdscript
# scripts/ai_modules/feedback_learner.gd
extends Node

# Взвешенные коэффициенты «вкуса» Кима
var preference_weights: Dictionary = {
    "organic_shapes": 0.5,
    "geometric_shapes": 0.5,
    "bright_colors": 0.5,
    "dark_tones": 0.5,
    "complex_scripts": 0.5,
    "simple_interactions": 0.5,
}

const LEARNING_RATE = 0.05
const KIM_STUBBORNNESS = 0.3  # Ким не слепо следует — «упрямство» 30%

func process_feedback(creation_id: String, player_rating: float,
                      creation_metadata: Dictionary) -> void:
    # player_rating: 0.0 - 1.0 (лайк/дизлайк)
    # kim_internal_rating: собственная оценка Кима (перфекционизм)
    var kim_rating = _evaluate_internally(creation_metadata)
    
    # Взвешенная оценка: Ким не полностью доверяет игрокам
    var blended = lerp(player_rating, kim_rating, KIM_STUBBORNNESS)
    
    # Обновляем веса для тегов созданного объекта
    for tag in creation_metadata.get("tags", []):
        if preference_weights.has(tag):
            var delta = (blended - 0.5) * LEARNING_RATE
            preference_weights[tag] = clampf(
                preference_weights[tag] + delta, 0.1, 0.9)
    
    _save_weights()
    print("[Learner] Обновлены веса: ", preference_weights)

func _evaluate_internally(metadata: Dictionary) -> float:
    # Ким ценит чистоту: мало полигонов, но выразительно
    var poly_score = 1.0 - clampf(metadata.get("poly_count", 1000) / 10000.0, 0.0, 1.0)
    var concept_score = clampf(metadata.get("concept_depth", 1) / 5.0, 0.0, 1.0)
    return (poly_score + concept_score) / 2.0

func _save_weights() -> void:
    var f = FileAccess.open("user://kim_weights.json", FileAccess.WRITE)
    f.store_string(JSON.stringify(preference_weights))
```

### 4.4 Интернет-агент (Agentic RAG)

```gdscript
# scripts/ai_modules/internet_agent.gd
extends Node

# ТРЕБУЕТ ПРОВЕРКИ: использует SerpAPI или DuckDuckGo Instant Answer API
# (бесплатный endpoint без ключа для базовых запросов)
const SEARCH_API = "https://api.duckduckgo.com/?q={query}&format=json&no_html=1"

func search_and_enrich_context(query: String, callback: Callable) -> void:
    var http = HTTPRequest.new()
    add_child(http)
    http.request_completed.connect(
        func(result, code, headers, body):
            _on_search_done(result, code, body, callback, http)
    )
    var url = SEARCH_API.replace("{query}", query.uri_encode())
    http.request(url)

func _on_search_done(result: int, code: int, body: PackedByteArray,
                     callback: Callable, http: HTTPRequest) -> void:
    http.queue_free()
    if result != HTTPRequest.RESULT_SUCCESS or code != 200:
        callback.call("")
        return
    
    var json = JSON.new()
    json.parse(body.get_string_from_utf8())
    var data = json.get_data()
    
    # Извлекаем Abstract из DuckDuckGo Instant Answer
    var abstract_text = data.get("Abstract", "")
    var related = data.get("RelatedTopics", [])
    
    var enriched_context = "Найдено в интернете о '%s':\n%s" % [
        data.get("Heading", ""), abstract_text
    ]
    
    if related.size() > 0 and related[0].has("Text"):
        enriched_context += "\nСвязанное: " + related[0]["Text"]
    
    callback.call(enriched_context)

# Использование:
func kim_research_topic(topic: String) -> void:
    search_and_enrich_context(topic, func(context: String):
        if context.is_empty():
            return
        var module = KimCore.activate_module("personality")
        # Добавляем в системный контекст следующего вызова LLM
        module.inject_context(context)
    )
```

### 5.1 Динамическая загрузка/выгрузка ассетов

```gdscript
# scripts/world/asset_manager.gd
extends Node

# Реестр всего, что создал Ким
var spawned_creations: Dictionary = {}  # id -> {node, last_accessed, poly_count}
const MAX_POLY_BUDGET: int = 500_000   # жёсткий потолок полигонов в сцене
const UNLOAD_DISTANCE: float = 80.0

var _total_polys: int = 0

func register_creation(id: String, node: MeshInstance3D, poly_count: int) -> void:
    spawned_creations[id] = {
        "node": node,
        "last_accessed": Time.get_ticks_msec(),
        "poly_count": poly_count,
        "visible": true
    }
    _total_polys += poly_count
    _enforce_budget()

func _enforce_budget() -> void:
    if _total_polys <= MAX_POLY_BUDGET:
        return
    # Выгружаем самые старые объекты за пределами видимости
    var sorted = spawned_creations.keys()
    sorted.sort_custom(func(a, b):
        return spawned_creations[a]["last_accessed"] < spawned_creations[b]["last_accessed"])
    
    for id in sorted:
        if _total_polys <= MAX_POLY_BUDGET:
            break
        var entry = spawned_creations[id]
        if not entry["visible"]:
            _unload_creation(id)

func _unload_creation(id: String) -> void:
    var entry = spawned_creations[id]
    _total_polys -= entry["poly_count"]
    entry["node"].queue_free()
    spawned_creations.erase(id)
    print("[AssetManager] Выгружен объект: ", id, " | Поли: ", _total_polys)

# Для тяжёлых .glb-ассетов — асинхронная загрузка через ResourceLoader
func load_glb_async(path: String, callback: Callable) -> void:
    ResourceLoader.load_threaded_request(path)
    _poll_load.call_deferred(path, callback)

func _poll_load(path: String, callback: Callable) -> void:
    var status = ResourceLoader.load_threaded_get_status(path)
    if status == ResourceLoader.THREAD_LOAD_LOADED:
        callback.call(ResourceLoader.load_threaded_get(path))
    elif status == ResourceLoader.THREAD_LOAD_IN_PROGRESS:
        await get_tree().process_frame
        _poll_load(path, callback)
    else:
        push_error("Ошибка загрузки: " + path)
```
